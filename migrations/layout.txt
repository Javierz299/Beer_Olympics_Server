CREATE TABEL "players"(
    "player_id" SERIAL PRIMARY KEY,
    "full_name" VARCHAR(32) NOT NULL,
    "nickname" VARCHAR(32) NOT NULL,
)

CREATE TABLE "teams"(
    "team_id" SERIAL PRIMARY KEY,
    "team_name" VARCHAR(32) NOT NULL,
    "player_name" VARCHAR(32) NOT NULL
    "player_id" INT references "players"(player_id) ON DELETE CASCADE NOT NULL
)

CREATE TABLE "games"(
    "game_id" SERIAL PRIMARY KEY,
    "team_name" VARCHAR references "teams"(team_name) ON DELETE CASCADE NOT NULL,
    "player_name VARCHAR references "teams"(player_name) ON DELETE CASCADE NOT NULL,
    "player_id" INT references "teams"(player_id) ON DELETE CASCADE NOT NULL
)
